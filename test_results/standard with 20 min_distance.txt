def get_constellation(input_audio, shouldPlot=False):
    spectrogram = np.abs(librosa.stft(input_audio, n_fft=1024, window='hann', win_length=1024, hop_length=512))
    constellation = peak_local_max(np.log(spectrogram), min_distance=20, threshold_rel=0.05, indices=False)
    if shouldPlot:
        plt.figure(figsize=(10, 5))
        plt.imshow(constellation, cmap=plt.cm.gray_r, origin='lower')
        plt.show()
    return constellation
